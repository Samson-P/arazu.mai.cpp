#include "pch.h"
#include <conio.h>
#include <iostream>
using namespace std;
const int w = 120, p = 28; // 79*79
bool gameover;
int x, y;

void drawmap() { // отрисовка карты

	for (int i = 1; i <= w; ++i) {
		cout << "H";
	}

	cout << endl;

	for (int i = 1; i < p; ++i) 
	{
		cout << "H";
		for (int j = 2; j < w; ++j)
		{
			if (i == y and j == x)
			{
				cout << "o";
			}
			else {
				cout << " ";
			}
		}
		cout << "H" << endl;
	}

	for (int i = 1; i <= w; ++i) {
		cout << "H";
	}
	cout << endl;

}

void setings() { // настройки
	x = rand() % w;
	y = rand() % p;
	bool gameover{ 0 };

}

void Input() {
	
	if (_kbhit()) {
		switch (_getch())
		{
		case 'a':
			if (x > 2) {
				x = x - 1;
			}
			break;


		case 'w':
			if (y > 1) {
				y = y - 1;
			}
			break;


		case 'd':
			if (x < w) {
				x = x + 1;
			}
			break;


		case 's':
			if (y < p - 1) {
				y = y + 1;
			}
			break;
		case 'x':
			gameover = 1;
			break;
		}
	}
}

void GO()
{
	system("cls");
	cout << "you lose! =3"; // написать прикольную заставку
}

int main()
{
	setings();
	while (!gameover)
	{
		system("cls");
		Input();
		drawmap();
	}

	GO();
}
